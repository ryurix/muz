{user} -- имя пользователя
{city} -- город пользователя
{adres} -- адрес пользователя
{adres!} -- предупреждение о разных адресах в заказах
{phone} -- Телефон
{login} -- Электропочта
{shop} -- Адрес для документов, берётся из свойств города
{staff} -- имя менеджера, если менеджер = покупатель, то пусто
{comment} -- комментарии к заказам, одной строкой через запятую

[*vendor*] -- колонка Поставщик
{vendors} -- список поставщиков через запятую

{rows_rus} -- количество строк прописью
{amount_rus} -- итоговое количество прописью
{total_rus} -- итоговая сумма прописью с копейками

[*money*] -- колонка Предоплата
{totalb} -- сумма предоплат
{totalb_rus} -- сумма предоплат прописью
{totalp} -- сумма с вычетом предоплаты
{totalp_rus} -- сумма с вычетом предоплат происью

{ptype} -- способ оплаты из пользователя
{dost} -- способ доставки из заказа

{uname} -- Юридическое название
{head} -- Директор
{uadr} -- Юр. адрес
{fadr} -- Физ. адрес
{inn} -- ИНН
{kpp} -- КПП
{okpo} -- ОКПО
{bank} -- банк
{bik} -- БИК банка
{bras} -- р/с банка
{bkor} -- к/с банка
{other} -- другая информация

Колонки в таблице:
[*store*] -- артикул: [*store1*], [*store1*], [*store1*]
[*names*] -- наименование
[*counts*] -- количество
[*pricem*] -- цена без налога
[*pricen*] -- налог на цену
[*prices*] -- цена с налогом
[*summam*] -- сумма без налога
[*summan*] -- сумма налога
[*summas*] -- сумма с налогом

{totalc} -- количество товаров на странице: {totalc1}, {totalc2}, {totalc3}
{totalm} -- сумма без налога на странице
{totaln} -- сумма налога на странице
{total} -- сумма с налогом на странице
